// ─── Phase 4: Budget Governor v3 Types ──────────────────────────────────────

/**
 * Cost provenance label — every budget metric must carry this.
 * Never conflate actual with estimated or unavailable.
 *
 * - actual:      real token/spend data from the provider API
 * - estimated:   pre-flight or modeled estimate; not confirmed
 * - unavailable: adapter cannot provide cost data (e.g. CLI bridge with no usage hook)
 */
export type CostProvenance = "actual" | "estimated" | "unavailable";

/**
 * Per-attempt budget preflight estimate produced before admission.
 * Used to gate attempts before any tokens are spent.
 */
export interface BudgetPreflightEstimate {
  /** Estimated input tokens for the compiled prompt. */
  estimatedPromptTokens: number;
  /** Overhead tokens for tool schemas and structured payloads. */
  estimatedToolOverheadTokens: number;
  /** Maximum output tokens the attempt is permitted to produce. */
  estimatedOutputTokensMax: number;
  /** Estimated USD cost of running verifier commands. */
  estimatedVerifierCostUsd: number;
  /** Total projected USD cost for this attempt. */
  estimatedAttemptCostUsd: number;
  /** Provenance of this estimate. Always "estimated" for preflight. */
  provenance: CostProvenance;
}

/**
 * Actual cost settlement written to the ledger after an attempt completes.
 * Separates patch cost from verification cost (R4.7).
 */
export interface BudgetSettlement {
  runId: string;
  attemptIndex: number;
  /** Actual patch/generation spend. Provenance label required. */
  patchCost: {
    usd: number;
    tokensIn: number;
    tokensOut: number;
    provenance: CostProvenance;
  };
  /** Actual verification spend (verifier commands, tool calls). */
  verificationCost: {
    usd: number;
    provenance: CostProvenance;
  };
  /** Combined total for this attempt. */
  totalActualUsd: number;
  /** Preflight estimate this attempt was admitted against. */
  preflightEstimateUsd: number;
  /** Variance between actual and estimate (positive = over-spent). */
  varianceUsd: number;
  settledAt: string;
}

// ─── Phase 4: Grounding Scanner ──────────────────────────────────────────────

export type GroundingViolationKind =
  | "file_not_found"
  | "symbol_not_found"
  | "import_not_found"
  | "patch_outside_allowed_paths";

export interface GroundingViolation {
  kind: GroundingViolationKind;
  reference: string;
  /** Source location hint from the patch (e.g. "diff hunk line 14"). */
  sourceHint?: string;
}

export interface GroundingScanResult {
  violations: GroundingViolation[];
  scannedReferences: number;
  /** Files referenced in the patch that were found in the index. */
  resolvedFiles: string[];
}

/**
 * Scan a unified diff string for file and symbol references, then validate
 * each against the repo grounding index.
 *
 * R4.1: Grounding scanner detects nonexistent files/symbols referenced in patches.
 *
 * @param diff  - Raw unified diff string (output of git diff or similar).
 * @param index - Repo grounding index from buildRepoGroundingIndex.
 * @param options.allowedPaths - If provided, flag references outside these globs as violations.
 */
export function scanPatchForGroundingViolations(
  diff: string,
  index: RepoGroundingIndex,
  options: { allowedPaths?: string[] } = {}
): GroundingScanResult {
  const violations: GroundingViolation[] = [];
  const resolvedFiles: string[] = [];
  const indexedPaths = new Set(index.files.map((f) => f.path));
  const indexedSymbols = new Set(index.files.flatMap((f) => f.symbols));

  // Extract file paths from diff headers (--- a/path, +++ b/path)
  const diffFilePattern = /^(?:---|\+\+\+)\s+[ab]\/(.+)$/gm;
  const referencedFiles = new Set<string>();
  for (const match of diff.matchAll(diffFilePattern)) {
    const p = match[1]?.trim();
    if (p && p !== "/dev/null") referencedFiles.add(p);
  }

  // Validate each referenced file against the index
  for (const filePath of referencedFiles) {
    if (indexedPaths.has(filePath)) {
      resolvedFiles.push(filePath);
    } else {
      violations.push({
        kind: "file_not_found",
        reference: filePath,
        sourceHint: `diff header references ${filePath}`
      });
    }

    // Check allowed paths if provided
    if (options.allowedPaths && options.allowedPaths.length > 0) {
      const isAllowed = options.allowedPaths.some((pattern) =>
        matchesGlobPattern(filePath, pattern)
      );
      if (!isAllowed) {
        violations.push({
          kind: "patch_outside_allowed_paths",
          reference: filePath,
          sourceHint: `${filePath} is outside allowed paths: ${options.allowedPaths.join(", ")}`
        });
      }
    }
  }

  // Extract import/require statements from added lines (+lines)
  const addedLinePattern = /^\+(?!\+\+)(.+)$/gm;
  const importPattern = /(?:import|require)\s*(?:.*\s+from\s+)?['"]([^'"]+)['"]/g;

  for (const lineMatch of diff.matchAll(addedLinePattern)) {
    const line = lineMatch[1] ?? "";
    for (const importMatch of line.matchAll(importPattern)) {
      const ref = importMatch[1] ?? "";
      // Only validate relative imports (starts with . or /) — skip npm packages
      if (!ref.startsWith(".") && !ref.startsWith("/")) continue;
      // Normalize: strip leading ./ and resolve to a plausible path
      const normalized = ref.replace(/^\.\//, "").replace(/\.(js|ts|tsx|jsx)$/, "");
      const hasMatch =
        index.files.some((f) => f.path.includes(normalized)) ||
        indexedSymbols.has(normalized);
      if (!hasMatch) {
        violations.push({
          kind: "import_not_found",
          reference: ref,
          sourceHint: `import in added line: ${line.trim().slice(0, 80)}`
        });
      }
    }
  }

  return {
    violations,
    scannedReferences: referencedFiles.size,
    resolvedFiles
  };
}

/**
 * Minimal glob matcher supporting ** and * wildcards.
 * Does not depend on external libraries — used only for allowedPaths checks.
 */
function matchesGlobPattern(filePath: string, pattern: string): boolean {
  // Convert glob to regex: ** = any path segment, * = no slash
  const regexStr = pattern
    .replace(/[.+^${}()|[\]\\]/g, "\\$&")
    .replace(/\*\*/g, "__DOUBLESTAR__")
    .replace(/\*/g, "[^/]*")
    .replace(/__DOUBLESTAR__/g, ".*");
  return new RegExp(`^${regexStr}$`).test(filePath);
}

// ─── Phase 4: Budget Preflight ───────────────────────────────────────────────

export interface BudgetPreflightInput {
  /** Approximate character count of the compiled prompt. */
  promptCharCount: number;
  /** Number of previous attempts in this loop (affects context size). */
  attemptCount: number;
  /** Remaining budget in USD. */
  remainingBudgetUsd: number;
  /** Per-attempt hard cap in USD (default: 20% of maxUsd). */
  perAttemptCapUsd?: number;
  /** Rough token price per 1M tokens for estimation (default: $3.00/1M). */
  pricePerMTokenUsd?: number;
}

export interface BudgetPreflightDecision {
  allowed: boolean;
  reason: string;
  estimate: BudgetPreflightEstimate;
}

/**
 * Computes a pre-admission cost estimate and applies admission gates:
 * - Total projected attempt cost must not exceed remaining budget
 * - Total projected attempt cost must not exceed per-attempt cap
 *
 * R4.4: Budget preflight runs before every attempt admission.
 * R4.5: Admission control rejects attempts that exceed budget band or per-attempt cap.
 */
export function evaluateBudgetPreflight(input: BudgetPreflightInput): BudgetPreflightDecision {
  const pricePerMToken = input.pricePerMTokenUsd ?? 3.0;

  // Estimate prompt tokens: ~4 chars per token, +20% for encoding overhead
  const rawPromptTokens = Math.ceil(input.promptCharCount / 4);
  const estimatedPromptTokens = Math.ceil(rawPromptTokens * 1.2);

  // Tool overhead scales with attempt count (more retries = larger tool schemas)
  const estimatedToolOverheadTokens = 800 + input.attemptCount * 200;

  // Output ceiling: allow up to 4k tokens per attempt
  const estimatedOutputTokensMax = 4_000;

  // Verifier cost: nominal $0.01 per attempt (verifier commands run locally)
  const estimatedVerifierCostUsd = 0.01;

  const totalInputTokens = estimatedPromptTokens + estimatedToolOverheadTokens;
  const estimatedAttemptCostUsd =
    roundUsd(
      ((totalInputTokens + estimatedOutputTokensMax) / 1_000_000) * pricePerMToken
    ) + estimatedVerifierCostUsd;

  const estimate: BudgetPreflightEstimate = {
    estimatedPromptTokens,
    estimatedToolOverheadTokens,
    estimatedOutputTokensMax,
    estimatedVerifierCostUsd,
    estimatedAttemptCostUsd,
    provenance: "estimated"
  };

  // Gate 1: remaining budget
  if (estimatedAttemptCostUsd > input.remainingBudgetUsd) {
    return {
      allowed: false,
      reason: `Preflight: estimated attempt cost $${estimatedAttemptCostUsd} exceeds remaining budget $${input.remainingBudgetUsd}.`,
      estimate
    };
  }

  // Gate 2: per-attempt cap (default 20% of remaining budget, minimum $0.05)
  const perAttemptCap = input.perAttemptCapUsd ?? Math.max(input.remainingBudgetUsd * 0.2, 0.05);
  if (estimatedAttemptCostUsd > perAttemptCap) {
    return {
      allowed: false,
      reason: `Preflight: estimated attempt cost $${estimatedAttemptCostUsd} exceeds per-attempt cap $${roundUsd(perAttemptCap)}.`,
      estimate
    };
  }

  return {
    allowed: true,
    reason: "Preflight passed.",
    estimate
  };
}

function roundUsd(value: number): number {
  return Math.round(value * 10_000) / 10_000;
}
