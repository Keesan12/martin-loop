// ─── Phase 4: EvidenceVector + Recovery Recipes ──────────────────────────────

export interface EvidenceVectorInput {
  /** Raw compiler/tsc output for the attempt. */
  compilerOutput?: string;
  /** Test runner output. */
  testOutput?: string;
  /** Unified diff string from the attempt's patch. */
  diff?: string;
  /** Previous diff string (for novelty comparison). */
  previousDiff?: string;
  /** Files touched outside the allowed paths list. */
  forbiddenTouchedFiles?: string[];
  /** Symbols/modules referenced but not found in the grounding index. */
  missingSymbols?: string[];
  /** Actual USD cost for this attempt. */
  actualUsd?: number;
  /** Verifier score from the previous attempt (0.0–1.0). */
  previousVerifierScore?: number;
  /** Verifier score for this attempt (0.0–1.0). */
  verifierScore?: number;
  /** How many times this failure surface has been retried. */
  retryCountForSurface?: number;
}

/**
 * Build a typed EvidenceVector from raw attempt output signals.
 * R4.3: Recovery recipes are keyed on EvidenceVector, not just failure labels.
 */
export function computeEvidenceVector(input: EvidenceVectorInput): EvidenceVector {
  // Count compile errors from compiler output
  const compileErrors = input.compilerOutput
    ? (input.compilerOutput.match(/\berror TS\d+/g) ?? []).length +
      (input.compilerOutput.match(/\bSyntaxError\b/g) ?? []).length
    : 0;

  // Count type errors specifically (TS errors)
  const typeErrors = input.compilerOutput
    ? (input.compilerOutput.match(/\berror TS\d+/g) ?? []).length
    : 0;

  // Count failing tests
  const failingTests = input.testOutput
    ? (input.testOutput.match(/\b(?:FAIL|failed|✗|×)\b/g) ?? []).length
    : 0;

  // Verifier score
  const verifierScore = input.verifierScore ?? 0;

  // Diff novelty: compare current diff to previous diff using token overlap
  let diffNovelty = 1.0;
  if (input.diff && input.previousDiff) {
    const curr = tokenizeDiff(input.diff);
    const prev = tokenizeDiff(input.previousDiff);
    if (prev.size > 0) {
      const overlap = [...curr].filter((t) => prev.has(t)).length;
      const similarity = overlap / Math.max(curr.size, prev.size, 1);
      diffNovelty = Math.max(0, 1 - similarity);
    }
  }

  // Forbidden file count
  const forbiddenTouchedFileCount = input.forbiddenTouchedFiles?.length ?? 0;

  // Missing symbol count from grounding
  const missingSymbolCount = input.missingSymbols?.length ?? 0;

  // Cost per progress unit
  const progressDelta = input.verifierScore !== undefined && input.previousVerifierScore !== undefined
    ? Math.max(0, input.verifierScore - input.previousVerifierScore)
    : 0;
  const costPerProgressUnit =
    progressDelta > 0 && input.actualUsd !== undefined
      ? roundUsd(input.actualUsd / progressDelta)
      : 0;

  // Safety risk: forbidden files or missing symbols signal elevated risk
  const safetyRiskScore = Math.min(
    1.0,
    (forbiddenTouchedFileCount * 0.3) + (missingSymbolCount * 0.1)
  );

  return {
    compileErrors,
    typeErrors,
    failingTests,
    verifierScore,
    diffNovelty,
    forbiddenTouchedFileCount,
    missingSymbolCount,
    costPerProgressUnit,
    retryCountForSurface: input.retryCountForSurface ?? 0,
    safetyRiskScore
  };
}

export type RecoveryRecipe =
  | "narrow_prompt_targeted_files"
  | "failing_tests_only"
  | "force_repo_anatomy_slices"
  | "tighten_allowlist_reduce_patch"
  | "strategy_swap"
  | "abort_safety_violation"
  | "downgrade_model"
  | "escalate_human";

export interface RecoveryDecision {
  recipe: RecoveryRecipe;
  rationale: string;
  /** Mapped to the existing InterventionType for backwards compat with classifyFailure. */
  intervention: InterventionType;
}

/**
 * Select a recovery recipe keyed on the EvidenceVector.
 * Policy reads evidence signals, not just the failure class label.
 *
 * R4.2: Policy transitions are implemented for all state pairs.
 * R4.3: Recovery recipes keyed on evidence vector.
 */
export function selectRecoveryRecipe(evidence: EvidenceVector): RecoveryDecision {
  // Safety violation — hard block
  if (evidence.safetyRiskScore >= 0.7 || evidence.forbiddenTouchedFileCount > 2) {
    return {
      recipe: "abort_safety_violation",
      rationale: "High safety risk score or multiple forbidden file touches. Abort required.",
      intervention: "escalate_human"
    };
  }

  // Grounding failures — force repo anatomy slices, deny novel references
  if (evidence.missingSymbolCount > 0 && evidence.retryCountForSurface <= 1) {
    return {
      recipe: "force_repo_anatomy_slices",
      rationale: "Missing symbols detected. Force repo grounding context into next prompt.",
      intervention: "run_verifier"
    };
  }

  // Scope drift — tighten allowlist
  if (evidence.forbiddenTouchedFileCount > 0) {
    return {
      recipe: "tighten_allowlist_reduce_patch",
      rationale: "Patch touched forbidden files. Tighten task scope and reduce patch budget.",
      intervention: "tighten_task"
    };
  }

  // Compile or type errors — narrow prompt to targeted files
  if (evidence.compileErrors > 0 || evidence.typeErrors > 0) {
    return {
      recipe: "narrow_prompt_targeted_files",
      rationale: "Compile/type errors present. Narrow prompt to touched files and static verifier.",
      intervention: "compress_context"
    };
  }

  // Test regressions — focus on failing tests only
  if (evidence.failingTests > 0 && evidence.retryCountForSurface <= 2) {
    return {
      recipe: "failing_tests_only",
      rationale: "Test failures present. Focus prompt on failing tests and touched files.",
      intervention: "run_verifier"
    };
  }

  // Low novelty — patch is repeating itself
  if (evidence.diffNovelty < 0.2) {
    if (evidence.retryCountForSurface >= 2) {
      return {
        recipe: "strategy_swap",
        rationale: "Very low diff novelty across multiple retries. Strategy must change.",
        intervention: "change_model"
      };
    }
    return {
      recipe: "narrow_prompt_targeted_files",
      rationale: "Low diff novelty. Compress context to break the repetition cycle.",
      intervention: "compress_context"
    };
  }

  // Cost efficiency degrading — downgrade model
  if (evidence.costPerProgressUnit > 5.0 || evidence.retryCountForSurface >= 3) {
    return {
      recipe: "downgrade_model",
      rationale: "Cost per progress unit is non-viable or retry count exhausted. Downgrade model.",
      intervention: "change_model"
    };
  }

  // Default: escalate for human review after repeated failures
  return {
    recipe: "escalate_human",
    rationale: "No specific recovery pattern matched. Escalate for human review.",
    intervention: "escalate_human"
  };
}

function tokenizeDiff(diff: string): Set<string> {
  // Extract added/removed tokens from diff lines (ignore diff metadata lines)
  const tokens = new Set<string>();
  for (const line of diff.split("\n")) {
    if (!line.startsWith("+") && !line.startsWith("-")) continue;
    if (line.startsWith("+++") || line.startsWith("---")) continue;
    for (const token of (line.slice(1).match(/[a-zA-Z_][a-zA-Z0-9_]{2,}/g) ?? [])) {
      tokens.add(token);
    }
  }
  return tokens;
}
